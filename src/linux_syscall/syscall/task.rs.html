    /// Exit the current thread group (i.e. process)
    pub fn sys_exit_group(&self, exit_code: usize) -> ! {
        let proc = self.zircon_process();
        info!("exit_group: code={}", exit_code);
        proc.exit(exit_code as i64);
        Thread::exit();
    }
